sg.js Documentation

sg.clamp(value, min, max)

Takes a value and keeps it within the given range.

return Math.min(Math.max(value, min), max); // Examples var a = 2; a = sg.clamp(a, 0, 1); // will set a to 1 var b = -1; b = sg.clamp(b, 0, 1); // will set b to 0

sg.randomRange(min, max)

Returns a random number within the given range.

return Math.random() * (max - min) + min; // Example var a = sg.randomRange(1, 5); // will set a to a random value from 1 to 5

sg.angDist(x1, y1, x2, y2)

Returns the angular distance between two points.

var xdif = Math.abs(x1 - x2); var ydif = Math.abs(y1 - y2); return Math.hypot(xdif, ydif); // Example var a = sg.angDist(0, 0, 1, 1) // will set a to ~1.414s

sg.ang(x1, y1, x2, y2)

Returns the angle between two points.

var yy = x2 - x1; // move it so x1 = y1 = 0 var xx = y2 - y1; var angle = Math.atan2(xx, yy); // angle from [0, 0] to [xx, yy] return angle; // Example var a = sg.ang(0, 0, 1, 1); // will set a to Ï€/2

sg.overshootInc(value, incriment, limit)

Adds to a value, keeping it from passing a limit.

var addition = value + incriment; // add them together var dir = Math.sign(incriment); if(dir == 1) { // cap to limit addition = Math.min(addition, limit); }else{ addition = Math.max(addition, limit); } return addition; // Examples var a = 0; a = sg.overshootInc(a, 10, 5); // will set a to 5 var b = 5; b = sg.overshootInc(b, -10, 0); // will set b to 0

sg.resize(canvas)

Resizes the canvas to fit the window.

canvas.style.position = "absolute"; // stops unessicary scrollbars canvas.width = window.innerWidth; canvas.height = window.innerHeight; // Example sg.resize(document.getElementsByTagName("canvas")[0])// will resize the first canvas on the document to fit the screen.

sg.lerp(value, target, ammount)

Returns an incriment toward the target by the spesified ammount.

return value + (target - value) * ammount; // Example var a = sg.lerp(1, 2, 0.5); // will set a to 1.5

sg.choose(...)

Returns a random agrument that was given.

return arguments[Math.floor(Math.random() * (arguments.length))]; // Example var a = sg.choose(1, 5, 3, "red", object, function(){console.log("hi")});

sg.onMobile()

Returns true if on a mobile device.

return /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent); // Example var a = sg.onMobile();
